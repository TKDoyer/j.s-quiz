Timer Example
--------------------

What is an interval?
An object that Javascript creates and manages for you that 
will do *something* that you tell it to over and over at a certain
interval

---- code ---
x = 0

x++ // 1
x == 2 // false
++x == 2 // true

---- memory ---
x = 2


let timerId;
function startTimer(duration, display) {
    let timer = duration;
    let minutes;
    let seconds;
    
    let timerTask = function() {
        minutes = parseInt(timer / 60, 10);
        seconds = parseInt(timer % 60, 10);

        minutes = minutes < 10 ? "0" + minutes : minutes;
        seconds = seconds < 10 ? "0" + seconds : seconds;

        display.textContent = minutes + ":" + seconds;
        x = 0
        
        timer = timer -1
        if (timer < 0) {
            timer = duration;
        }
        
        if (timer == 0) {
            timer = "0:00"
        }
    }
    timerId = setInterval(timerTask, 1000);
}

function resetTimer() {
        clearInterval(timerId);
}

   
function start10Timer() {
    var tenMinutes = 60 * 10,
    display = document.querySelector('#time10');
    startTimer(tenMinutes, display);
};
